# Quick API example
import requests

# Create a solution
response = requests.post(
    "http://localhost:8000/solutions/",
    json={
        "name": "Banking Fraud Detection",
        "solution_type": "research",  # Fast & free
        "workflows": ["Fraud Detection Workflow"],
        "tools": ["fraud_detector", "transaction_lookup"]
    }
)

solution_id = response.json()['id']

# Execute fraud check
result = requests.post(
    f"http://localhost:8000/solutions/{solution_id}/execute",
    json={
        "workflow_name": "Fraud Detection Workflow",
        "inputs": {
            "transaction_data": {
                "amount": 5000,
                "location": "OVERSEAS",
                "hour": 2
            }
        }
    }
)

print(f"Fraud Score: {result.json()['fraud_score']}")
print(f"Processing Time: <100ms")
print(f"Cost: $0.00")
